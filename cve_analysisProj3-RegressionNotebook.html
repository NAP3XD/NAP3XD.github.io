def load_cve_data(year):
    """Load CVE data for a specific year"""
    filename = f'data/nvdcve-2.0-{year}.json.gz'
    try:
        with gzip.open(filename, 'rt', encoding='utf-8') as f:
            data = json.load(f)
            return data.get('vulnerabilities', [])
    except Exception as e:
        print(f"Error loading {filename}: {e}")
        return []

def extract_vendors_from_cpe(cve_data):
    """Extract vendor names from CPE strings"""
    vendors = set()
    if 'configurations' in cve_data:
        for config in cve_data.get('configurations', []):
            for node in config.get('nodes', []):
                for cpe_match in node.get('cpeMatch', []):
                    criteria = cpe_match.get('criteria', '')
                    # CPE format: cpe:2.3:a:vendor:product:version...
                    parts = criteria.split(':')
                    if len(parts) > 3:
                        vendor = parts[3]
                        if vendor != '*':
                            vendors.add(vendor)
    return list(vendors)

def parse_datetime(date_str):
    """Parse datetime string from CVE data"""
    if not date_str:
        return None
    try:
        return datetime.strptime(date_str, '%Y-%m-%dT%H:%M:%S.%f')
    except:
        try:
            return datetime.strptime(date_str, '%Y-%m-%dT%H:%M:%S')
        except:
            return None

def get_cvss_score(cve_data):
    """Extract CVSS base score"""
    metrics = cve_data.get('metrics', {})
    if 'cvssMetricV31' in metrics and len(metrics['cvssMetricV31']) > 0:
        return metrics['cvssMetricV31'][0]['cvssData'].get('baseScore', None)
    if 'cvssMetricV30' in metrics and len(metrics['cvssMetricV30']) > 0:
        return metrics['cvssMetricV30'][0]['cvssData'].get('baseScore', None)
    if 'cvssMetricV2' in metrics and len(metrics['cvssMetricV2']) > 0:
        return metrics['cvssMetricV2'][0]['cvssData'].get('baseScore', None)
    return None

def get_attack_vector(cve_data):
    """Extract attack vector from CVSS metrics"""
    metrics = cve_data.get('metrics', {})
    if 'cvssMetricV31' in metrics and len(metrics['cvssMetricV31']) > 0:
        return metrics['cvssMetricV31'][0]['cvssData'].get('attackVector', None)
    if 'cvssMetricV30' in metrics and len(metrics['cvssMetricV30']) > 0:
        return metrics['cvssMetricV30'][0]['cvssData'].get('attackVector', None)
    return None

def get_cwe_type(cve_data):
    """Extract CWE type"""
    weaknesses = cve_data.get('weaknesses', [])
    if weaknesses:
        descriptions = weaknesses[0].get('description', [])
        if descriptions:
            return descriptions[0].get('value', None)
    return None

# Parse all CVE data
cve_records = []
for vuln in all_vulnerabilities:
    cve = vuln.get('cve', {})
    if cve.get('vulnStatus') == 'Rejected':
        continue
    published = parse_datetime(cve.get('published'))
    last_modified = parse_datetime(cve.get('lastModified'))
    if not published or not last_modified:
        continue
    time_to_patch = (last_modified - published).total_seconds() / 86400
    if time_to_patch <= 0:
        continue
    vendors = extract_vendors_from_cpe(cve)
    cvss_score = get_cvss_score(cve)
    attack_vector = get_attack_vector(cve)
    cwe_type = get_cwe_type(cve)
    descriptions = cve.get('descriptions', [])
    desc_length = len(descriptions[0].get('value', '')) if descriptions else 0
    ref_count = len(cve.get('references', []))
    if vendors:
        for vendor in vendors:
            record = {
                'cve_id': cve.get('id'),
                'vendor': vendor,
                'published': published,
                'last_modified': last_modified,
                'time_to_patch_days': time_to_patch,
                'cvss_score': cvss_score,
                'attack_vector': attack_vector,
                'cwe_type': cwe_type,
                'description_length': desc_length,
                'reference_count': ref_count,
                'year': published.year,
                'month': published.month,
                'day_of_week': published.weekday()
            }
            cve_records.append(record)
df = pd.DataFrame(cve_records)
print(f"Created dataset with {len(df):,} records")
